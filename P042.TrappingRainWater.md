# 42. Trapping Rain Water
Given n non-negative integers representing an elevation map where the width of each bar is 1, compute how much water it is able to trap after raining.

For example, 
Given [0,1,0,2,1,0,1,3,2,1,2,1], return 6.

class Solution {
public:
    void test()
    {
        vint nums = { 0,1,0,2,1,0,1,3,2,1,2,1 };
        cout << trap(nums) << endl;
        PRINT1D(nums, " ");
    }
    int trap(vector<int>& height) {
        int res = 0;
        while (1)
        {
            int tmp = trap_(height);
            res += tmp;
            if (tmp == 0) break;
        }
        return res;
    }

    int trap_(vector<int>& height) {
        if (height.size() <= 1) return 0;
        auto N = height.size();
        vint dif(N);
        dif[0] = 1;
        for (int i = 1; i < N ; i++)
        {
            dif[i] = height[i] - height[i - 1] > 0 ? 1 : 0;
        }
        vint vex;
        for (int i = 0; i < N ; i++)
        {
            if (dif[i] == 1 && (i + 1 == N || dif[i + 1] == 0))
            {
                vex.push_back(i);
            }
                
        }
        if (vex.size() <= 1) return 0;
        int res = 0;
        int left = height[vex.front()];
        int mid = left;
        
        for (int i = 1; i < vex.size() ; i++)
        {
            int right = height[vex[i]];
            mid = MMIN(left, right);
            for (int j = vex[i - 1]+1; j < vex[i]; ++j) {
                int tmp = mid - height[j];
                if (tmp > 0) {
                    res += tmp;
                    height[j] = mid;
                }
            }
            left = right;
        }
        return res;
    }
};

int main()

    Solution so;
    so.test();
    return 0;
