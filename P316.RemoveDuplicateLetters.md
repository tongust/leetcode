# 316 Remove Duplicate Letters
Given a string which contains only lowercase letters, remove duplicate letters so that every letter appear once and only once. You must make sure your result is the smallest in lexicographical order among all possible results.

Example:
Given "bcabc"
Return "abc"

Given "cbacdcbc"
Return "acdb"


```cpp
#include <string>
#include <iostream>
#include <vector>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>
#include <list>

using namespace std;


class Solution {
public:
    typedef unordered_map<char/*val*/, unordered_set<int>/*idx*/> maptype;
    typedef unordered_set<int> setinttype;
    std::tuple<maptype, list<int>> get_map(string &s) {
        maptype mmap;
        list<int> bold;
        std::tuple<maptype, decltype(bold)> res;
        for (int i = 0; i < s.size(); ++i) {
            char c = s[i];
            if (mmap.find(c) != mmap.end()) {
                mmap[c].insert(i);
            }
            else
            {
                setinttype mset;
                mset.insert(i);
                mmap[c] = mset;
            }
        }
        
        maptype mmap1;
        int mc = -1;
        for (auto c : s) {
            mc++;
            if (mmap[c].size() == 1) {
                bold.push_back(mc);
            }
            else {
                mmap1[c] = mmap[c];
            }
        }
        res = std::make_tuple(mmap1, bold);
        return res;
    }
    template<typename T, typename C>
    bool isFind(T &mmap, C c) {
        return mmap.find(c) != mmap.end();
    }
    string removeDuplicateLetters(string s) {
        s.push_back(127);
        maptype mmap;
        list<int> bold;
        std::tie(mmap, bold) = get_map(s);
        
        int pre = -1;
        int sz_bold = bold.size();
        auto it_bold = bold.begin();
        int sz_mmap = mmap.size();
        while (bold.size() < sz_bold + sz_mmap)
        {
            bool case0_occur = false,
                case1_occur = false,
                case2_occur = false,
                case3_occur = false;
            int idx_min = *it_bold;
            char c_min = s[idx_min];

            // spacial case
            maptype tmpmap;
            if (pre+1 == *it_bold) {
                pre = *it_bold;
                it_bold++;
                continue;
            }
            for (int i = pre + 1; i < *it_bold; ++i) {
                char c = s[i];
                if (isFind(tmpmap, c)) {
                    tmpmap[c].insert(i);
                }
                else {
                    setinttype tss;
                    tss.insert(i);
                    tmpmap[c] = tss;

                }
                    
            }
            int idx_min1 = 0;
            vector<char> mins;
            for (auto &it : tmpmap) {
                if (isFind(mmap, it.first) && it.second.size() == mmap[it.first].size()) {
                    mins.push_back(it.first);
                    if (it.second.size() == 1) {
                        case3_occur = true;
                        break;
                    }
                    // idx_min1 = *std::min_element(it.second.begin(), it.second.end());
                    // pre = pre;
                    case2_occur = true;
                    // break;
                }
            }
            if (case3_occur) {
                char e11 = mins.back();
                int eidx = *mmap[e11].begin();
                it_bold = bold.insert(it_bold, eidx);
                mmap.erase(e11);
                continue;
            }
            if (case2_occur) {
                auto minc = *std::min_element(mins.begin(), mins.end());
                idx_min1 = *std::max_element(mmap[minc].begin(), mmap[minc].end());
                char erase_c = minc;
                int erase_idx = idx_min1;
                for (int i = pre + 1; i < idx_min1; ++i) {
                    if (isFind(mmap, s[i])) {
                        if (s[i] < erase_c) {
                            erase_c = s[i];
                            erase_idx = i;
                        }
                    }
                }
                if (erase_idx == idx_min1) {
                    idx_min1 = *std::min_element(mmap[minc].begin(), mmap[minc].end());
                    it_bold = bold.insert(it_bold, idx_min1);
                    mmap.erase(s[idx_min1]);
                    continue;
                }
                else {
                    it_bold = bold.insert(it_bold, erase_idx);
                    mmap.erase(s[erase_idx]);
                    continue;
                }
                
            }
            for (int i = pre + 1; i < *it_bold; ++i) {
                char c = s[i];
                if (isFind(mmap, c)) {
                    // case 0
                    if (mmap[c].size() == 1) {
                        it_bold = bold.insert(it_bold, i);
                        // pre = pre;
                        case0_occur = true;
                        break;
                    }
                    // case 1
                    else {
                        if (c_min > c) {
                            c_min = c;
                            idx_min = i;
                            case1_occur = true;
                        }
                    }
                }
            }
            
            if (case1_occur && !case0_occur) {
                mmap.erase(s[idx_min]);
                for (int i = pre + 1; i < idx_min; ++i) {
                    char c = s[i];
                    if (isFind(mmap, c)) {
                        mmap[c].erase(i);
                    }
                }
                pre = idx_min;
                bold.insert(it_bold, idx_min);
                continue;
            }
            else if (!case0_occur && !case1_occur) {
                
                for (int i = pre + 1; i < *it_bold; ++i) {
                    char c = s[i];
                    if (isFind(mmap, c)) {
                        mmap[c].erase(i);
                    }
                }
                pre = *it_bold;
                it_bold++;
                continue;
            }
            else if (case0_occur)
            {
                pre = pre;
                continue;
            }
        }
        string res(bold.size(), ' ');
        auto it0 = bold.begin();
        for (int i = 0; i < res.size(); ++i) {
            res[i] = s[*it0];
            it0++;
        }
        res.pop_back();
        return res;
    }
};
```
